/* 
Fn to print any int Array:
*/

#include<stdio.h>

void piarray(int tcolumns,int array[][tcolumns]){//only total no of columns needed as argument
    
    for(int i =0;array[i][0]!='\0';i++){
        
        for(int j=0;array[i][j]!='\0';j++){
            
            printf("%d ",array[i][j]);
            
        }
        
    }
    
    
}




int main(){
    
    int array[5][5]={1,1,2,3,54,6,67,4,56,7};
    
    piarray(5,array);
    
    
    return 0;
    
}
